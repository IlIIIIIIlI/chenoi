def KMEANS(data, class_label):
    data = data[data['class'] == class_label]
    scaler =  MinMaxScaler()
    scaler_data = scaler.fit_transform(data.drop(['class', 'sub_class'], axis=1))
    # 使用kmeans
    clf = KMeans(n_clusters=5, random_state=2022)
    pred = clf.fit(scaler_data)
    return pred.labels_
